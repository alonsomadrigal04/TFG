[DIALOGUE COMMANDS]
·   [b] text [/b]                                   -- Bold text
·   [i] text [/i]                                   -- Italic text
·   [w] text [/w]                                   -- Text with a wavy effect
·   [s] text [/s]                                   -- Text with a shaking effect
·   [speed=number] text [/speed]                    -- Text that appears at a specific speed (e.g., [speed=50]Hello[/speed])
·   [c=color] text [/c]                             -- Text with a specific color (e.g., [c=red]Hello[/c])
[TYPE COMMANDS]
·   SAY                                             -- A character says a line of dialogue
·   SAY/question                                    -- A small question mark appears next to the character
·   SAY/exclamation                                 -- A small exclamation mark appears next to the character
·   CHOICE/Soft                                     -- A choice appears with a soft transition
·   CHOICE/Impact                                   -- A choice appears with a Impact transition
[CODE COMMANDS]
// POSITIONS //
LEFT, CENTER, RIGHT, FARRIGHT, FARLEFT
// CHARACTERS //
·   [NAME OF CHARACTER] appears [POSITIONS]         -- A character appears in a position on the screen
·   [NAME OF CHARACTER] disappears                  -- A character disappears from the screen
·   [NAME OF CHARACTER] moves [POSITIONS]           -- A character moves to a position on the screen
·   [NAME OF CHARACTER] moves [POSITIONS] noflip    -- A character moves to a position on the screen without flipping their sprite
·   [NAME OF CHARACTER] Is [EMOTION]                -- A character changes their emotion
·   [NAME OF CHARACTER] flips                       -- A character changes the orientation of their sprite
// BACKGROUNDS //
·   [NAME OF BACKGROUND] bg                         -- A background appears on the screen
·   [NAME OF BACKGROUND] trans [TYPE]               -- A background transition occurs with a specific type (e.g., flash, blur)
·   [NAME OF BACKGROUND] flashback                  -- A flashback effect occurs with a specific duration (e.g., flashback 3s)
// CAMERA //
·   Camera zoom [POSITION/CHARACTER] [DURATION]     -- The camera zooms in or out on a position or character for a specific duration (e.g., Camera zoom Center 2s)
·   Camera reset                                    -- The camera resets to its default position and zoom level
·   Camera shake [DURATION] [INTENSITY]             -- The camera shakes for a specific duration (e.g., Camera shake 1s)
·   Camera flash [DURATION] [INTENSITY]             -- The camera flashes for a specific duration (e.g., Camera flash 0.5s)
// OBJECT //
·   [NAME OF OBJECT] shows                          -- An object appears at the top right corner of the screen
// GAME STATE //
·   [NAME] affinity [VALUE]                         -- Adds or subtracts affinity points of a character (e.g., Alonso affinity 10)
·   [NAME]_affinity > [VALUE] : [UID NAME]          -- If a character's affinity is greater than a specific value, jump to a specific UID (e.g., Alonso_affinity > 50 : [UID1])                                
·   [NAME]_affinity < [VALUE] : [UID NAME]          -- If a character's affinity is less than a specific value, jump to a specific UID (e.g., Alonso_affinity < 20 : [UID2])
·   [NAME] needs [OBJECT] : [UID NAME]              -- If the player has a specific object, jump to a the following line. If not, jump to a specific UID (e.g., Alonso needs Key : [UID3])  
// TIME //
·   Wait [DURATION]                                 -- The game waits for a specific duration before proceeding (e.g., Wait 2s)
[UID COMMANDS]
·  CHAPTER ENDS                                     -- Ends the current chapter and moves to the next one
·  END GAME                                         -- Ends the game and returns to the main menu
·  GO TO DIALOG [DIALOG NAME]                       -- Jumps to a specific dialog in the gamedata
·  GO TO EXPLOREZONE [EXPLORE NAME]                 -- Jumps to a specific explore scene in the gamedata
·  [empty]                                          -- Moves to the next line of dialogue
·  [UID NAME]                                       -- Jumps to a specific UID in the same dialog (e.g., [UID1])
